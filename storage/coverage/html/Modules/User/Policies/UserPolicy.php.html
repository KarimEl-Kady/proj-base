7/**
8 * Record-level authorization for User records. Route middleware
9 * (`permission:users.update`, `permission:users.delete` — see
10 * Routes/api.php) gates the *capability*; a Policy gates the *record*, once
11 * a specific target user is known. This is the base's canonical example of
12 * that split — copy this pattern for any module whose actions need more
13 * than "does the actor hold this permission".
14 *
15 * Auto-discovered by Laravel via the Models -> Policies namespace
16 * convention (Gate::guessPolicyName replaces "\Models\" with "\Policies\"),
17 * so nothing registers this class explicitly. Permission's Gate::before
18 * hook only ever *allows* (see PermissionServiceProvider::registerGateHook)
19 * and defers otherwise, so it steps aside cleanly for the record-specific
20 * logic below.
21 */
22class UserPolicy
23{
24    /**
25     * A user may always update their own profile — PROJECT_AUTH_DEFAULT_ROLE
26     * grants no permissions by default, so without this, a freshly
27     * registered user could never change their own name or password.
28     */
29    public function update(User $actor, User $subject): bool
30    {
31        return $actor->is($subject) || $actor->hasPermissionTo('users.update');
32    }
33
34    /**
35     * Never allow self-deletion through this endpoint, even for an admin
36     * holding users.delete: an admin (or an attacker riding a hijacked
37     * session) locking the account out is a support incident, not a
38     * feature. An account-deletion flow a user runs on themselves, if the
39     * product needs one, deserves its own deliberately separate endpoint
40     * (re-auth prompt, confirmation, etc.) — not this one.
41     */
42    public function delete(User $actor, User $subject): bool
43    {
44        if ($actor->is($subject)) {
45            return false;
46        }
47
48        return $actor->hasPermissionTo('users.delete');
49    }
50}
