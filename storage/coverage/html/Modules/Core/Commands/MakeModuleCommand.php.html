778    protected function cleanContent(string $content): string
779    {
780        $lines = explode("\n", $content);
781
782        if (count($lines) <= 1) {
783            return $content;
784        }
785
786        $minIndent = PHP_INT_MAX;
787        foreach ($lines as $line) {
788            if (trim($line) === '') {
789                continue;
790            }
791            if (preg_match('/^(\s+)/', $line, $m)) {
792                $minIndent = min($minIndent, strlen($m[1]));
793            } else {
794                $minIndent = 0;
795                break;
796            }
797        }
798
799        if ($minIndent > 0 && $minIndent < PHP_INT_MAX) {
800            foreach ($lines as &$line) {
801                if ($line !== '') {
802                    $line = substr($line, $minIndent);
803                }
804            }
805        }
806
807        return implode("\n", $lines)."\n";
808    }
