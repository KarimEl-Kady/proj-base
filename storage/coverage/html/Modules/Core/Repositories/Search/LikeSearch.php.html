7/**
8 * The default: `WHERE column LIKE '%word%'` OR'd across every searchable
9 * column. Zero setup, correct, and fine for most modules — most tables in
10 * a project this size never cross into the row counts where this matters.
11 *
12 * The known tradeoff: a leading wildcard can't use a B-tree index, so this
13 * degrades to a full scan per searchable column as a table grows into the
14 * millions of rows. When a specific module's table actually gets there,
15 * override BaseRepository::searchStrategy() in that repository with
16 * FullTextSearch (paired with a real fulltext index migration) rather than
17 * changing this default for every module that doesn't need it.
18 */
19class LikeSearch implements SearchStrategy
20{
21    public function apply(Builder $query, string $word, array $columns): Builder

23        // LIKE wildcards in the search word are user data, not operators.
